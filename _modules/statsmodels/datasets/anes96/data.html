def _get_data():
    filepath = dirname(abspath(__file__))
    data = recfromtxt(open(filepath + '/anes96.csv',"rb"), delimiter="\t",
            names = True, dtype=float)
    logpopul = log(data['popul'] + .1)
    data = nprf.append_fields(data, 'logpopul', logpopul, usemask=False,
                                                          asrecarray=True)
    return data
