        """
        Score vector for Arma model

        #return None

        jac = ndt.Jacobian(self.loglike, stepMax=1e-4)
        return jac(params)[-1]

        """
        Hessian of arma model.  Currently uses numdifftools
        """
        #return None
        Hfun = ndt.Jacobian(self.score, stepMax=1e-4)
        return Hfun(params)[-1]
